compile

javac -d . Simple.java

jar cvf simple.jar com/outside/Simple.class

-----------------------------------------------------------

mvn install:install-file -Dfile=K:\simple.jar -DgroupId=david -DartifactId=husen -Dversion=0.1 -Dpackaging=jar
